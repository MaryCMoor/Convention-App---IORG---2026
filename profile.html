async function saveProfile() { const profile = JSON.parse(localStorage.getItem(`profile_${currentUser.username}`) || '{}'); profile.name = document.getElementById('profileName').value; profile.email = document.getElementById('profileEmail').value; profile.phone = document.getElementById('profilePhone').value; profile.age = document.getElementById('profileAge').value; profile.assembly = document.getElementById('profileAssembly').value; // Emergency contact profile.emergencyName = document.getElementById('emergencyName').value; profile.emergencyPhone = document.getElementById('emergencyPhone').value; profile.emergencyRelation = document.getElementById('emergencyRelation').value; // Dietary restrictions profile.dietVegetarian = document.getElementById('dietVegetarian').checked; profile.dietVegan = document.getElementById('dietVegan').checked; profile.dietGlutenFree = document.getElementById('dietGlutenFree').checked; profile.dietDairyFree = document.getElementById('dietDairyFree').checked; profile.dietOther = document.getElementById('dietOther').value; // Accessibility profile.accessWheelchair = document.getElementById('accessWheelchair').checked; profile.accessHearing = document.getElementById('accessHearing').checked; profile.accessVisual = document.getElementById('accessVisual').checked; profile.accessOther = document.getElementById('accessOther').value; localStorage.setItem(`profile_${currentUser.username}`, JSON.stringify(profile)); // Update currentUser in localStorage currentUser.name = profile.name; currentUser.email = profile.email; localStorage.setItem('currentUser', JSON.stringify(currentUser)); console.log('=== SAVE PROFILE STARTED ==='); // Prepare data for Google Sheets const sheetData = { action: 'saveUser', userId: currentUser.username, username: currentUser.username, email: profile.email || '', name: profile.name || '', role: (profile.roles || [profile.role || 'User']).join(';'), password: profile.password || '', dateCreated: profile.dateCreated || new Date().toISOString() }; console.log('Data to send to Google Sheets:', sheetData); // Save to Google Sheets try { console.log('Sending fetch request...'); const response = await fetch(SCRIPT_URL, { method: 'POST', mode: 'no-cors', // ← ADDED THIS! redirect: 'follow', // ← ADDED THIS! headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(sheetData) }); console.log('Fetch completed'); console.log('Response status:', response.status); console.log('Response type:', response.type); alert('✅ Profile saved successfully!'); } catch (error) { console.error('Error saving to Google Sheets:', error); alert('✅ Profile saved locally!\n\n⚠️ Warning: Could not sync to Google Sheets.\nError: ' + error.message); } }
